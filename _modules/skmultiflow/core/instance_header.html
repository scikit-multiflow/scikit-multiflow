        """ get_header_label_at

        Gets the header label at index header_index.

        Parameters
        ----------
        header_index: int
            An index from the instance header.

        Returns
        -------
        string
            The index label.


        """
        if (header_index > -1) and (header_index < len(self.header)):
            return self.header[header_index]
        else:
            return None
